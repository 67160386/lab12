### ✏️1.1 อธิบายความแตกต่างระหว่าง state ธรรมดากับ useState Hook

เขียนตัวอย่างโค้ด 2 แบบเพื่อแสดงความแตกต่าง:

- แบบ 1: ตัวแปรธรรมดา (ไม่ใช้ useState)
- แบบ 2: ตัวแปร state (ใช้ useState)
  แล้วอธิบายว่าทำไม UI ในแบบ 2 จึงอัปเดตอัตโนมัติ
  - ตัวแปรธรรมดา: ค่าเปลี่ยน แต่ React ไม่รู้เรื่อง จึงไม่วาดหน้าจอใหม่
  - useState: ค่าเปลี่ยนและบอก React ให้วาดหน้าจอใหม่ (Re-render)

```jsx
// แบบ 1: ตัวแปรธรรมดา (ไม่ใช้ useState)
// ในแบบนี้ แม้ค่าใน Memory จะเปลี่ยนไปแล้ว แต่หน้าจอ (UI) จะ "นิ่งสนิท" เหมือนเดิม
function Counter() {
  let count = 0; // ตัวแปรธรรมดา

  const increment = () => {
    count = count + 1;
    console.log("ค่าปัจจุบันใน Memory:", count);
    // ค่าใน Console จะเพิ่มขึ้นเรื่อยๆ แต่หน้าจอจะแสดงเลข 0 ตลอดไป
  };

  return (
    <div>
      <p>คะแนน: {count}</p>
      <button onClick={increment}>เพิ่มคะแนน</button>
    </div>
  );
}

// แบบ 2: ตัวแปร state (ใช้ useState)
// แบบนี้คือวิธีมาตรฐานของ React เมื่อค่าเปลี่ยน หน้าจอจะอัปเดตตามทันที
import { useState } from "react";

function CounterWithState() {
  const [count, setCount] = useState(0); // ใช้ State Hook

  const increment = () => {
    setCount(count + 1); // ใช้ฟังก์ชันเพื่อบอก React ให้เปลี่ยนค่า
  };

  return (
    <div>
      <p>คะแนน: {count}</p>
      <button onClick={increment}>เพิ่มคะแนน</button>
    </div>
  );
}
```

### ✏️1.2 เมื่อต้องการจัดการ state แบบ Object (เช่น user ที่มี name, email, age) ควรเขียนแบบไหน?

เขียนโค้ดสั้นๆ ที่แสดง:

- การสร้าง state เป็น Object
- การอัปเดต property หนึ่งของ Object โดยไม่ทำให้ property อื่นหายไป
- อธิบายว่าเหตุใด spread operator จำเป็น
  - ใน React การอัปเดต State แบบ Object จะเป็นการ "แทนที่ (Replace)" ไม่ใช่การ "แก้ไข (Patch)
  - การใช้ ... คือการเอาข้อมูลที่มีอยู่ทั้งหมดมาด้วย แล้วจะขอเปลี่ยนแค่บางอย่างเท่านั้น

```jsx
import { useState } from "react";

function UserProfile() {
  // 1. สร้าง State เป็น Object
  const [user, setUser] = useState({
    name: "John",
    email: "john@example.com",
    age: 25,
  });

  const updateEmail = () => {
    // 2. อัปเดตเฉพาะ email โดยใช้ Spread Operator (...)
    setUser({
      ...user, // คัดลอกค่าเดิมทั้งหมดมาวาง (name, age)
      email: "new@mail.com", // เขียนทับเฉพาะตัวที่ต้องการเปลี่ยน
    });
  };

  return (
    <div>
      <p>
        ชื่อ: {user.name} | อีเมล: {user.email} | อายุ: {user.age}
      </p>
      <button onClick={updateEmail}>เปลี่ยนอีเมล</button>
    </div>
  );
}
```

### ✏️2.2 เมื่อสร้าง Form ใน React ควรใช้ Controlled Component ทำไม?

อธิบายพร้อมเขียนตัวอย่างโค้ด input ที่เป็น controlled component

```jsx
import { useState } from "react";

function MyForm() {
  const [name, setName] = useState(""); // 1. สร้าง State มารอรับค่า

  const handleChange = (e) => {
    const newValue = e.target.value;
    // เราสามารถใส่ Logic ตรวจสอบตรงนี้ได้เลย
    if (newValue.length <= 10) {
      setName(newValue); // 2. อัปเดต State เมื่อมีการพิมพ์
    }
  };

  return (
    <div>
      <label>ชื่อ (ไม่เกิน 10 ตัวอักษร): </label>
      <input
        type="text"
        value={name} // 3. ผูกค่าในช่อง Input เข้ากับ State
        onChange={handleChange} // 4. เมื่อพิมพ์ ให้ไปสั่งเปลี่ยน State
      />
      <p>สิ่งที่คุณพิมพ์คือ: {name}</p>
    </div>
  );
}
```

- Ternary Operator - แสดง/ซ่อน 2 element
- Logical AND (&&) - แสดง element เมื่อเงื่อนไขเป็น true
- if/else ภายนอก JSX - ใช้ variable เก็บ element

### ✏️3.2 เขียน Component ที่แสดงสถานะ Loading, Success, Error

- เมื่อ status = "idle" แสดง "กรุณากดปุ่ม"
- เมื่อ status = "loading" แสดง "⏳ กำลังโหลด..."
- เมื่อ status = "success" แสดง "สำเร็จ!"
- เมื่อ status = "error" แสดง "❌ เกิดข้อผิดพลาด"

```jsx
import { useState } from "react";

function StatusManager() {
  // 1. ใช้ useState เก็บ status โดยเริ่มที่ "idle"
  const [status, setStatus] = useState("idle");

  // 2. ใช้ if/else ภายนอก JSX เพื่อเตรียมข้อมูลที่จะแสดง
  let message;
  let messageClass = "";

  if (status === "idle") {
    message = "กรุณากดปุ่ม";
  } else if (status === "loading") {
    message = "⏳ กำลังโหลด...";
  } else if (status === "success") {
    message = "สำเร็จ!";
    messageClass = "text-success";
  } else if (status === "error") {
    message = "❌ เกิดข้อผิดพลาด";
    messageClass = "text-error";
  }

  return (
    <div style={{ padding: "20px", textAlign: "center" }}>
      {/* 3. แสดงข้อความตามเงื่อนไขที่คำนวณไว้ข้างต้น */}
      <h2 className={messageClass}>{message}</h2>

      <hr />

      {/* 4. ปุ่มสำหรับเปลี่ยน status ต่างๆ */}
      <div style={{ display: "flex", gap: "10px", justifyContent: "center" }}>
        <button onClick={() => setStatus("loading")}>เริ่มโหลด</button>
        <button onClick={() => setStatus("success")}>จำลองสำเร็จ</button>
        <button onClick={() => setStatus("error")}>จำลองข้อผิดพลาด</button>
        <button onClick={() => setStatus("idle")}>รีเซ็ต</button>
      </div>
    </div>
  );
}
```

## 📋คำถาม 4: Form & State Management

### ✏️4.1 เขียน Component LoginForm ที่มี:

- Input: username
- Input: password
- ปุ่ม: ส่งข้อมูล
- แสดงข้อความ "เข้าสู่ระบบสำเร็จเป็น [username]" เมื่อส่งเสร็จ

```jsx
import { useState } from "react";

function LoginForm() {
  // 1. จัดการข้อมูล Input ทั้งหมดด้วย Object เดียว
  const [credentials, setCredentials] = useState({
    username: "",
    password: "",
  });

  // 2. จัดการสถานะการล็อกอิน
  const [isLoggedIn, setIsLoggedIn] = useState(false);

  // ฟังก์ชันจัดการการเปลี่ยนแปลงใน Input
  const handleChange = (e) => {
    const { name, value } = e.target;
    setCredentials({
      ...credentials, // กระจายค่าเดิม (Spread Operator)
      [name]: value, // อัปเดตเฉพาะ field ที่พิมพ์
    });
  };

  // ฟังก์ชันเมื่อกดส่งข้อมูล
  const handleSubmit = (e) => {
    e.preventDefault(); // ป้องกัน Browser รีโหลดหน้าเว็บ
    if (credentials.username && credentials.password) {
      setIsLoggedIn(true);
    } else {
      alert("กรุณากรอกข้อมูลให้ครบถ้วน");
    }
  };

  // 3. Conditional Rendering: ถ้าล็อกอินสำเร็จให้แสดงข้อความยืนยัน
  if (isLoggedIn) {
    return (
      <div style={{ textAlign: "center", marginTop: "20px" }}>
        <h2>เข้าสู่ระบบสำเร็จเป็น {credentials.username} ✅</h2>
        <button onClick={() => setIsLoggedIn(false)}>ออกจากระบบ</button>
      </div>
    );
  }

  return (
    <div
      style={{
        maxWidth: "300px",
        margin: "20px auto",
        padding: "20px",
        border: "1px solid #ccc",
      }}
    >
      <h3>Login</h3>
      <form onSubmit={handleSubmit}>
        <div style={{ marginBottom: "10px" }}>
          <input
            type="text"
            name="username"
            placeholder="Username"
            value={credentials.username}
            onChange={handleChange}
            style={{ width: "100%" }}
          />
        </div>
        <div style={{ marginBottom: "10px" }}>
          <input
            type="password"
            name="password"
            placeholder="Password"
            value={credentials.password}
            onChange={handleChange}
            style={{ width: "100%" }}
          />
        </div>
        <button type="submit" style={{ width: "100%", cursor: "pointer" }}>
          ส่งข้อมูล
        </button>
      </form>
    </div>
  );
}
```

### ✏️4.2 เขียน Component ที่มี Checkbox ติดต่อได้

```jsx
function NewsletterCheckbox() {
```

### ✏️5.2 เขียน Component ที่มี Todo List พร้อม:

- Input สำหรับเพิ่ม todo
- ปุ่มเพิ่ม
- รายการ todo พร้อมปุ่มลบแต่ละรายการ
- แสดง "ยังไม่มีรายการ" เมื่อไม่มี todo

```jsx
import { useState } from "react";

function TodoApp() {
  // 1. State สำหรับเก็บรายการ Todo (Array) และข้อความใน Input (String)
  const [todos, setTodos] = useState([]);
  const [inputValue, setInputValue] = useState("");

  // 2. ฟังก์ชันเพิ่มรายการใหม่
  const addTodo = (e) => {
    e.preventDefault(); // ป้องกันหน้าเว็บ Refresh เมื่อกดปุ่มใน Form
    if (inputValue.trim() === "") return; // ไม่เพิ่มถ้าช่องว่าง

    const newTodo = {
      id: Date.now(), // ใช้ Timestamp เป็น ID ชั่วคราว
      text: inputValue,
    };

    setTodos([...todos, newTodo]); // สร้าง Array ใหม่โดยใช้ Spread Operator
    setInputValue(""); // ล้างช่อง Input หลังเพิ่มเสร็จ
  };

  // 3. ฟังก์ชันลบรายการ
  const deleteTodo = (id) => {
    // กรองเอาตัวที่ ID ไม่ตรงออกเพื่อสร้าง Array ใหม่
    const updatedTodos = todos.filter((todo) => todo.id !== id);
    setTodos(updatedTodos);
  };

  return (
    <div style={{ padding: "20px", maxWidth: "400px", margin: "auto" }}>
      <h2>รายการสิ่งที่ต้องทำ 📝</h2>

      {/* ส่วน Input: Controlled Component */}
      <form onSubmit={addTodo} style={{ marginBottom: "20px" }}>
        <input
          type="text"
          value={inputValue}
          onChange={(e) => setInputValue(e.target.value)}
          placeholder="เพิ่มรายการใหม่..."
          style={{ padding: "8px", width: "70%" }}
        />
        <button type="submit" style={{ padding: "8px", marginLeft: "5px" }}>
          เพิ่ม
        </button>
      </form>

      {/* ส่วนแสดงผล: Conditional Rendering & Mapping */}
      {todos.length === 0 ? (
        <p style={{ color: "gray" }}>ยังไม่มีรายการ</p>
      ) : (
        <ul style={{ listStyle: "none", padding: 0 }}>
          {todos.map((todo) => (
            <li
              key={todo.id}
              style={{
                display: "flex",
                justifyContent: "space-between",
                padding: "10px",
                borderBottom: "1px solid #eee",
              }}
            >
              {todo.text}
              <button
                onClick={() => deleteTodo(todo.id)}
                style={{
                  backgroundColor: "#ff4d4d",
                  color: "white",
                  border: "none",
                  borderRadius: "4px",
                  cursor: "pointer",
                }}
              >
                ลบ
              </button>
            </li>
          ))}
        </ul>
      )}
    </div>
  );
}
```
